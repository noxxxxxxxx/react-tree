# Guide

# Quick Start

Prerequisites

  • Familiarity with the React and Redux
  • Install Node.js version 16.0 or higher

# Install

Use NPM:

$ npm i figma-tree

Use Yarn:

$ yarn add figma-tree

Use PNPM:

$ pnpm i figma-tree

# Import

import Tree from 'figma-tree'
import 'figma-tree/dist/style.css'

export default App () => {
  const data = []
  const config = {}

  return (
    return (
    <Tree
      {...config}
      data={data}
    />
  )
  )
}
